28 class LiveWeatherHttp(using httpClient: HttpClient) extends WeatherHttp:
29   private def endpoint(latitude: Double, longitude: Double): URI =
30     val baseUrl = "https://api.open-meteo.com/v1/forecast"
31     URI.create(s"$baseUrl?latitude=$latitude&longitude=$longitude&current_weather=true")

33   def fetch(latitude: Double, longitude: Double): Try[String] =
34     val request = HttpRequest.newBuilder(endpoint(latitude, longitude)).GET().build()

37 class WeatherClient(using http: WeatherHttp, logger: Logger):
38   def get(latitude: Double, longitude: Double): List[String] =
39     http.fetch(latitude, longitude) match
40       case Failure(fetchErr) =>
41         logger.error(s"${fetchErr.getClass.getSimpleName}: ${fetchErr.getMessage}")
42         List(s"We had an error fetching the weather data. Please try again later.")
43       case Success(fetchBody) =>
44         logger.debug(s"Fetched weather data: $fetchBody")
45         Try(read[WeatherResponse](fetchBody)) match
46           case Failure(parseErr) =>
47             logger.error(s"Parse failure: ${parseErr.getMessage} when parsing: $fetchBody")
48             List(s"We had an error parsing the weather data. Please try again later.")
49           case Success(parsed) =>
50             parsed.currentWeather pipe { weather => List(
51               f"Current temperature: ${weather.temperatureC}%.1f C",
52               f"Wind speed: ${weather.windSpeedKmh}%.1f km/h",
53               s"Wind direction: ${weather.windDirectionDeg} degrees"
54             ) }
